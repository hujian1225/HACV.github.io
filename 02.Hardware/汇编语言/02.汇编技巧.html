  • C++

  • 演示如下：
#pragma comment(linker,"/STACK:1024000000,1024000000")
//1024后面跟上6个0

#include<iostream>
using namespace std;

# 2.2.Linux的g++编译器

  • g++
  • 在main函数里面写如下C语言内嵌汇编指令
int _size=256<<20;	//256MB
char *p=(char *)malloc(_size)+_size;
__asm__
(
"movl %0,%%esp\n"
:
:"r"(p)
);
