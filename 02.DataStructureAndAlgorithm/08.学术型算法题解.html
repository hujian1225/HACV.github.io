      # 1.1.382. 链表随机节点『纯模板』

      • 382. 链表随机节点
      • 选取k=1的蓄水池抽样『模板代码』
      /**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    /** @param head The linked list's head.
        Note that the head is guaranteed to be not null, so it contains at least one node. */
    Solution(ListNode* head) {
        inner_head=head;
    }
    
    /** Returns a random node's value. */
    int getRandom() {
        int selectRes=0x3f3f3f;//表示不选择
        //『易错的地方，一定要用临时变量，不能修改inner_head，我也不知道为啥』
        ListNode * loop=inner_head; 
        int curNum=1;
        // rand() % k == 0 模拟概率为 1/k 的情况
        while( nullptr!=loop )
        {
            if( 0==rand()%curNum )
            {
                selectRes=loop->val;
            }
            ++curNum;
            loop=loop->next;
        }

        return selectRes;
    }
private:
    ListNode * inner_head;
};

 * Solution* obj = new Solution(head);
 * int param_1 = obj->getRandom();

      # 1.2.398. 随机数索引

      • 398. 随机数索引
      • 纯模板的改进
      class Solution {
public:
    Solution(vector<int>& nums) :inner_nums(nums) {

    }
    
    int pick(int target) {
        int count=1;
        int res=0;  //假装是第0号
        for(int loop=0; loop<inner_nums.size(); ++loop)
        {
            //和纯模板不同的地方，只管那个我们想要管的
            if( target==inner_nums[loop] )
            {
                // rand() % k == 0 模拟概率为 1/k 的情况
                if( 0==rand()%count )
                {
                    res=loop;
                }
                ++count;
            }
        }

        return res;
    }
private:
    vector<int> inner_nums;
};

/**
 * Your Solution object will be instantiated and called as such:
 * Solution* obj = new Solution(nums);
 * int param_1 = obj->pick(target);
