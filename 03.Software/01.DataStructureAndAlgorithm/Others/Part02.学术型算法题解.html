    # 1.1.382. 链表随机节点『纯模板』

    • 382. 链表随机节点
    • 选取k=1的蓄水池抽样『模板代码』
    /**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    /** @param head The linked list's head.
        Note that the head is guaranteed to be not null, so it contains at least one node. */
    Solution(ListNode* head) {
        inner_head=head;
    }
    
    /** Returns a random node's value. */
    int getRandom() {
        int selectRes=0x3f3f3f;//表示不选择
        //『易错的地方，一定要用临时变量，不能修改inner_head，我也不知道为啥』
        ListNode * loop=inner_head; 
        int curNum=1;
        // rand() % k == 0 模拟概率为 1/k 的情况
        while( nullptr!=loop )
        {
            if( 0==rand()%curNum )
            {
                selectRes=loop->val;
            }
            ++curNum;
            loop=loop->next;
        }

        return selectRes;
    }
private:
    ListNode * inner_head;
};

    # 1.2.398. 随机数索引

    • 398. 随机数索引
    • 纯模板的改进
    class Solution {
public:
    Solution(vector<int>& nums) :inner_nums(nums) {

    }
    
    int pick(int target) {
        int count=1;
        int res=0;  //假装是第0号
        for(int loop=0; loop<inner_nums.size(); ++loop)
        {
            //和纯模板不同的地方，只管那个我们想要管的
            if( target==inner_nums[loop] )
            {
                // rand() % k == 0 模拟概率为 1/k 的情况
                if( 0==rand()%count )
                {
                    res=loop;
                }
                ++count;
            }
        }

        return res;
    }
private:
    vector<int> inner_nums;
};

    # 02.拒绝采样

    • 拒绝采样
    在数值分析和计算统计中，拒绝采样是用于从分布生成观测值的基本技术。
它通常也被称为接受拒绝方法或「接受拒绝算法」，是一种精确的模拟方法。该方法适用于具有密度的任何分布。

拒绝采样基于以下观察：在一维中对随机变量进行采样，可以对二维笛卡尔图执行均匀随机采样，并将样本保持在其密度函数图下的区域中。

拒绝采样是蒙特卡罗方法的一种，在众多的采样方法中，这种方法显得独树一帜。而且，这种方法与Metropolis-Hastings算法也有一定关系

    # 2.1.470. 用 Rand7() 实现 Rand10()

    • 470. 用 Rand7() 实现 Rand10()

    // The rand7() API is already defined for you.
// int rand7();
// @return a random integer in the range 1 to 7

class Solution {
public:
    int rand10() {
        int row, col, idx;
        do {
            row = rand7();
            col = rand7();
            idx = col + (row - 1) * 7;
        } while (idx > 40);
        return 1 + (idx - 1) % 10;
    }
};

    # 3.1.1352. 最后 K 个数的乘积

    • 1352. 最后 K 个数的乘积
    • 坑点：可能有0
    • 『奇怪的技巧学习』+前缀乘
    class ProductOfNumbers {
public:
    ProductOfNumbers() {
        LastBlockLen=0;
        NoZeroVec.resize(40000+5);
        NoZeroVec[0]=1;
    }
    
    void add(int num) {
        if( 0==num )
        {
            LastBlockLen=0;
        }
        else
        {
            NoZeroVec[ ++LastBlockLen ]=num;
            NoZeroVec[ LastBlockLen ]*=NoZeroVec[ LastBlockLen-1 ];
        }
    }
    
    int getProduct(int k) {
        if( k>LastBlockLen )
        {
            return 0;
        }
        else
        {
            //注意，也包含了特判『LastBlockLen==k』
            return NoZeroVec[ LastBlockLen ]/ NoZeroVec[ LastBlockLen-k ];
        }
    }
private:
    int LastBlockLen;
    vector<int> NoZeroVec;
};
